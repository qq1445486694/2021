在script代码中间添加debugger可以暂时停止下面代码的运行

	new Promise(function (resolve,reject) {
		console.log('one');
		resolve(2222);
	}).then(function (va) {
		console.log(va);
		return 1231;
	}).then(function (vvv) {
	    console.log(vvv)
	})
	function givepromise(msg,delay) {
		return new Promise(function (resolve,reject){
			setTimeout(function (argument) {
				console.log(msg);
				resolve();
			},delay)
		})
	}
	givepromise(100,1100).then(function (argument) {
		return givepromise(200,1200)
	})
